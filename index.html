                document.getElementById('toggleBoostBtn').addEventListener('click', () => this.toggleBoost());

                document.getElementById('resetGame').addEventListener('click', () => this.resetGame());

                document.getElementById('boostAmount').addEventListener('change', () => this.updateSettings());

                

                // Audio scrubber events

                document.getElementById('audioScrubber').addEventListener('input', (e) => this.scrubAudio(e));

                document.getElementById('audioScrubber').addEventListener('mousedown', () => this.isScrubbing = true);

                document.getElementById('audioScrubber').addEventListener('mouseup', () => this.isScrubbing = false);

            }


            async handleFileUpload(event) {

                const file = event.target.files[0];

                if (!file) return;


                document.getElementById('fileName').textContent = file.name;

                

                try {

                    console.log('Loading audio file:', file.name);

                    const arrayBuffer = await file.arrayBuffer();

                    this.audioBuffer = await this.audioContext.decodeAudioData(arrayBuffer);

                    console.log('Audio file loaded successfully:', {

                        duration: this.audioBuffer.duration,

                        sampleRate: this.audioBuffer.sampleRate,

                        channels: this.audioBuffer.numberOfChannels

                    });

                    

                    // Enable and update scrubber

                    const scrubber = document.getElementById('audioScrubber');

                    scrubber.disabled = false;

                    scrubber.max = this.audioBuffer.duration;

                    

                    // Update time display

                    document.getElementById('totalTime').textContent = this.formatTime(this.audioBuffer.duration);

                    document.getElementById('currentTime').textContent = '0:00';

                    

                    this.audioPosition = 0;

                } catch (error) {

                    console.error('Error loading audio file:', error);

                    alert('Error loading audio file. Please try a different file format (MP3, WAV, etc.).');

                    document.getElementById('fileName').textContent = '';

                }

            }


            generateFrequencyButtons() {

                const grid = document.getElementById('frequencyGrid');

                grid.innerHTML = '';

                

                const frequencies = this.levels[this.currentLevel];

                

                frequencies.forEach(freq => {

                    const button = document.createElement('button');

                    button.className = 'frequency-button';

                    button.textContent = this.formatFrequency(freq);

                    button.addEventListener('click', () => this.selectFrequency(freq, button));

                    grid.appendChild(button);

                });

                

                console.log('Generated buttons for frequencies:', frequencies);

            }


            formatFrequency(freq) {

                if (freq >= 1000) {

                    return `${freq / 1000}kHz`;

                }

                return `${freq}Hz`;

            }


            async playAudio() {

                if (!this.audioBuffer) {

                    alert('Please upload an audio file first');

                    return;

                }


                try {

                    // Resume audio context if suspended

                    if (this.audioContext.state === 'suspended') {

                        await this.audioContext.resume();

                    }


                    // Stop any existing audio

                    this.stopAudio();

                    

                    // Select random frequency for this round

                    this.selectRandomFrequency();

                    

                    // Set up and start audio playback

                    this.setupAudioNodes();

                    this.startTimeUpdater();

                    

                    // Update UI

                    const playBtn = document.getElementById('playBtn');

                    const toggleBtn = document.getElementById('toggleBoostBtn');

                    

                    playBtn.classList.add('active');

                    playBtn.textContent = 'Playing';

                    toggleBtn.disabled = false;

                    toggleBtn.textContent = 'Toggle EQ';

                    toggleBtn.classList.remove('active');

                    

                    this.isPlaying = true;

                    this.isBoostActive = false;

                    

                    console.log('Audio started playing successfully for frequency:', this.currentFrequency);

                } catch (error) {

                    console.error('Error playing audio:', error);

                    alert('Error playing audio. Please check your audio file and try again.');

                    this.stopAudio();

                }

            }


            setupAudioNodes() {

                try {

                    // Create source node

                    this.sourceNode = this.audioContext.createBufferSource();

                    this.sourceNode.buffer = this.audioBuffer;

                    this.sourceNode.loop = true; // Enable looping


                    // Create gain node for volume control

                    this.gainNode = this.audioContext.createGain();

                    this.gainNode.gain.value = 0.7;


                    // Create filter node for EQ boost

                    this.filterNode = this.audioContext.createBiquadFilter();

                    this.filterNode.type = 'peaking';

                    this.filterNode.frequency.value = this.currentFrequency;

                    this.filterNode.Q.value = 2;

                    this.filterNode.gain.value = 0; // Start with no boost


                    // Connect nodes

                    this.sourceNode.connect(this.filterNode);

                    this.filterNode.connect(this.gainNode);

                    this.gainNode.connect(this.audioContext.destination);


                    // Handle source end (shouldn't happen with looping, but just in case)

                    this.sourceNode.onended = () => {

                        console.log('Source ended unexpectedly');

                        if (this.isPlaying) {

                            // Restart playback if we're still supposed to be playing

                            setTimeout(() => {

                                if (this.isPlaying) {

                                    console.log('Restarting audio playback');

                                    this.setupAudioNodes();

                                }

                            }, 100);

                        }

                    };


                    // Start playback from current position with looping enabled

                    this.audioStartTime = this.audioContext.currentTime;

                    this.sourceNode.start(0, this.audioPosition);

                    

                    console.log(`Audio playback started with looping enabled for frequency: ${this.currentFrequency}Hz at position: ${this.audioPosition.toFixed(2)}s`);

                } catch (error) {

                    console.error('Error setting up audio nodes:', error);

                    alert('Error playing audio. Please try again.');

                    this.stopAudio();

                }

            }


            toggleBoost() {

                if (!this.isPlaying || !this.filterNode) {

                    console.log('Cannot toggle boost - not playing or no filter node');

                    return;

                }

                

                try {

                    const boostAmount = parseFloat(document.getElementById('boostAmount').value);

                    const toggleBtn = document.getElementById('toggleBoostBtn');

                    

                    this.isBoostActive = !this.isBoostActive;

                    this.filterNode.gain.setValueAtTime(

                        this.isBoostActive ? boostAmount : 0, 

                        this.audioContext.currentTime

                    );

                    

                    if (this.isBoostActive) {

                        toggleBtn.classList.add('active');

                        toggleBtn.textContent = 'EQ ON';

                    } else {

                        toggleBtn.classList.remove('active');

                        toggleBtn.textContent = 'Toggle EQ';

                    }

                    

                    console.log(`Boost toggled ${this.isBoostActive ? 'ON' : 'OFF'} at ${this.currentFrequency}Hz with ${boostAmount}dB`);

                } catch (error) {

                    console.error('Error toggling boost:', error);

                }

            }


            stopAudio() {

                try {

                    if (this.sourceNode) {

                        this.sourceNode.stop();

                        this.sourceNode.disconnect();

                        this.sourceNode = null;

                    }

                    

                    if (this.filterNode) {

                        this.filterNode.disconnect();

                        this.filterNode = null;

                    }

                    

                    if (this.gainNode) {

                        this.gainNode.disconnect();

                        this.gainNode = null;

                    }

                    

                    if (this.updateInterval) {

                        clearInterval(this.updateInterval);

                        this.updateInterval = null;

                    }


                    const playBtn = document.getElementById('playBtn');

                    const toggleBtn = document.getElementById('toggleBoostBtn');

                    

                    playBtn.classList.remove('active');

                    playBtn.textContent = 'Play';

                    toggleBtn.disabled = true;

                    toggleBtn.textContent = 'Toggle EQ';

                    toggleBtn.classList.remove('active');

                    

                    this.isPlaying = false;

                    this.isBoostActive = false;

                    

                    console.log('Audio stopped successfully');

                } catch (error) {

                    console.error('Error stopping audio:', error);

                }

            }


            startTimeUpdater() {

                this.updateInterval = setInterval(() => {

                    if (this.isPlaying && this.audioBuffer && !this.isScrubbing) {

                        const elapsed = this.audioContext.currentTime - this.audioStartTime;

                        this.audioPosition = (this.audioPosition + elapsed) % this.audioBuffer.duration;

                        this.audioStartTime = this.audioContext.currentTime;

                        

                        // Update scrubber position

                        document.getElementById('audioScrubber').value = this.audioPosition;

                        document.getElementById('currentTime').textContent = this.formatTime(this.audioPosition);

                    }

                }, 100);

            }

            

            scrubAudio(event) {

                if (!this.audioBuffer) return;

                

                this.audioPosition = parseFloat(event.target.value);

                document.getElementById('currentTime').textContent = this.formatTime(this.audioPosition);

                

                // If playing, restart from new position

                if (this.isPlaying) {

                    this.stopAudio();

                    setTimeout(() => {

                        this.playAudio();

                    }, 50);

                }

            }

            

            formatTime(seconds) {

                const mins = Math.floor(seconds / 60);

                const secs = Math.floor(seconds % 60);

                return `${mins}:${secs.toString().padStart(2, '0')}`;

            }


            selectRandomFrequency() {

                const frequencies = this.levels[this.currentLevel];

                this.currentFrequency = frequencies[Math.floor(Math.random() * frequencies.length)];

                console.log('Target frequency:', this.currentFrequency);

            }


            selectFrequency(frequency, buttonElement) {

                if (!this.isPlaying) {

                    alert('Please play the audio first');

                    return;

                }


                this.stopAudio();

                

                // Reset all button states

                document.querySelectorAll('.frequency-button').forEach(btn => {

                    btn.classList.remove('correct', 'incorrect');

                });


                if (frequency === this.currentFrequency) {

                    buttonElement.classList.add('correct');

                    this.correctCount++;

                    

                    if (this.correctCount >= 5) {

                        this.levelUp();

                    }

                } else {

                    buttonElement.classList.add('incorrect');

                    

                    // Show correct answer

                    document.querySelectorAll('.frequency-button').forEach(btn => {

                        if (btn.textContent === this.formatFrequency(this.currentFrequency)) {

                            btn.classList.add('correct');

                        }

                    });

                    

                    this.wrongCount++;

                    

                    if (this.wrongCount >= 3) {

                        this.gameOver();

                    }

                }

                

                this.updateDisplay();

                

                // Clear button states after 2 seconds

                setTimeout(() => {

                    document.querySelectorAll('.frequency-button').forEach(btn => {

                        btn.classList.remove('correct', 'incorrect');

                    });

                }, 2000);

            }


            levelUp() {

                if (this.currentLevel < 6) {

                    this.currentLevel++;

                    this.correctCount = 0;

                    this.wrongCount = 0;

                    

                    document.getElementById('newLevel').textContent = this.currentLevel;

                    document.getElementById('levelUpModal').style.display = 'block';

                    

                    this.updateDisplay();

                    this.generateFrequencyButtons();

                } else {

                    alert('Congratulations! You\'ve completed all levels!');

                    this.resetGame();

                }

            }


            gameOver() {

                document.getElementById('finalLevel').textContent = this.currentLevel;

                document.getElementById('gameOverModal').style.display = 'block';

            }


            resetGame() {

                this.stopAudio();

                this.currentLevel = 1;

                this.correctCount = 0;

                this.wrongCount = 0;

                this.updateDisplay();

                this.generateFrequencyButtons();

                this.closeModal('gameOverModal');

            }


            updateDisplay() {

                document.getElementById('currentLevel').textContent = this.currentLevel;

                document.getElementById('levelNumber').textContent = this.currentLevel;

                document.getElementById('correctCount').textContent = this.correctCount;

                document.getElementById('wrongCount').textContent = this.wrongCount;

                

                const progressPercent = (this.correctCount / 5) * 100;

                document.getElementById('progressFill').style.width = `${progressPercent}%`;

                

                const frequencies = this.levels[this.currentLevel].map(f => this.formatFrequency(f)).join(', ');

                document.getElementById('levelInstructions').textContent = 

                    `Play audio, toggle the EQ boost on/off, and identify which frequency is emphasized: ${frequencies}`;

            }


            updateSettings() {

                // Settings are applied in real-time during playback

                if (this.isPlaying && this.filterNode) {

                    const boostAmount = parseFloat(document.getElementById('boostAmount').value);

                    this.filterNode.gain.value = this.isBoostActive ? boostAmount : 0;

                }

                console.log('Settings updated');

            }

            }

        }
